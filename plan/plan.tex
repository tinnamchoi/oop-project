  \section{Description}
  A turn-based role-playing Pokémon™ fighting game. 
  
  \lipsum[1]
  
  \pagebreak

  \section{Potential classes}
  
  \subsection{Player}
  Data: 
  \begin{itemize}
    \item Pokemon owned
    \item Level 
  \end{itemize}
  Functions:
  \begin{itemize}
    \item Add/remove Pokemon owned
    \item Increment/decrement level
  \end{itemize}
  
  \subsubsection{Person}
  Data:
  \begin{itemize}
    \item Name
    \item Skill points (for leveling up)
  \end{itemize}
  Functions:
  \begin{itemize}
    \item Set name
    \item Increment/decrement skill points
    \item Get action from user
  \end{itemize}
  
  \subsubsection{Computer}
  Functions:
  \begin{itemize}
    \item Get action randomly
  \end{itemize}
    
  \subsection{Pokemon}
  Data:
  \begin{itemize}
    \item Name
    \item Type
    \item Level
    \item Health
    \item Attack
    \item Defense
    \item Speed
    \item Moves learnt
  \end{itemize}
  Functions:
  \begin{itemize}
    \item Increment/decrement level
    \item Increment/decrement health
    \item Increment/decrement attack
    \item Increment/decrement defense
    \item Increment/decrement speed
    \item Add/remove moves learnt
  \end{itemize}
  
  \subsection{Menu}
  Data:
  \begin{itemize}
    \item Title
    \item Options (vector)
  \end{itemize}
  Functions:
  \begin{itemize}
    \item Print menu
    \item Set title
    \item Set options
  \end{itemize}
